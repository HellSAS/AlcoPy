60 def calculate_alcohol_by_volume(drink_type, drink_amount, weight, height, gender, fullness) -> float:
61 """! Рассчитывает уровень алкоголя в крови [грамм спирта/литр крови].
62 @param drink_type Тип напитка.
63 @param drink_amount Количество выпитого напитка (в миллилитрах).
64 @param weight Вес человека (в килограммах).
65 @param height Рост человека (в сантиметрах).

67 @param fullness Уровень сытости человека.
68 @return [грамм.спирта/литр.крови].
69 """
70
71 k = 1 if fullness == False else 10/17
72 bmi = weight/((height/100)**2)
73 a = drink_amount*AlcoCalc.drink_types[drink_type]["alcohol_content"]*0.789
74 rm = 1.0181-0.0113*bmi
75 rf = 0.9367-0.01240*bmi
76 r = rm if gender == "male" else rf
77 bac = a/(weight*r)*k
